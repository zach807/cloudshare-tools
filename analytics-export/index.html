Analytics Export Tool

Export all data from CloudShare's Sisense BI dashboards — training, sales enablement, experiences, resources, environments, and project members — as a single ZIP of CSVs.

What You'll Get

📚 Training Analytics

Course data, student records, participation, session time, drop-off rates

training_analytics.csv

💼 Sales Enablement

POC opportunities, status, SE ownership, usage hours, disk/GB metrics

sales_enablement.csv

🎯 Sales Experiences

Experience names, participants, status, active time, start/end dates

sales_experiences.csv

⚙️ Resource Consumption

Daily billing, project/blueprint/env breakdown, compute hours, storage

resources.csv

🖥️ Environments

Environment data from CloudShareEnvironmentsCube (auto-discovers fields)

environments.csv

👥 Project Members

Project member usage from ProjectMemberUsageCube (auto-discovers fields)

project_members.csv
How to Use
1

Open CloudShare Analytics

Log into CloudShare Accelerate, then click Analytics in the navigation. This opens bi.cloudshare.com in a new tab — stay on that tab.

2

Open the browser console

Press F12 (or Cmd+Option+J on Mac) to open DevTools, then click the Console tab. You must be on the bi.cloudshare.com page.

3

Copy & paste the script below

Copy the entire script, paste it into the console, and press Enter. Progress updates appear in the console as each dataset is fetched. A ZIP file downloads automatically when complete.

⚠️ Chrome security tip: If Chrome shows a warning about pasting, type allow pasting first and press Enter, then paste the script.

var http = window.prism.$injector.get('$http');
var errors = [];

// ---- Utility: build CSV from row arrays ----
function buildCSV(headers, rows) {
  var lines = [headers.join(',')];
  for (var i = 0; i < rows.length; i++) {
    var cells = rows[i].map(function(v) {
      var s = (v === null || v === undefined) ? '' : String(v);
      if (s.indexOf(',') !== -1 || s.indexOf('"') !== -1 || s.indexOf('\n') !== -1) {
        s = '"' + s.replace(/"/g, '""') + '"';
      }
      return s;
    });
    lines.push(cells.join(','));
  }
  return lines.join('\n');
}

// ---- Utility: extract cell value ----
function cellVal(cell) {
  if (!cell) return '';
  return cell.text !== undefined ? cell.text : (cell.data !== undefined ? cell.data : '');
}

// ---- Utility: JAQL query ----
function jaqlQuery(datasource, metadata) {
  return http.post('/api/datasources/' + datasource + '/jaql', {
    datasource: { title: datasource },
    metadata: metadata,
    count: 50000,
    offset: 0
  });
}

// ---- Utility: download ZIP (pure JS, no dependencies) ----
function downloadZip(files) {
  // Mini ZIP builder
  function toBytes(str) {
    var arr = [];
    for (var i = 0; i < str.length; i++) {
      var c = str.charCodeAt(i);
      if (c < 128) { arr.push(c); }
      else if (c < 2048) { arr.push(192 | (c >> 6), 128 | (c & 63)); }
      else { arr.push(224 | (c >> 12), 128 | ((c >> 6) & 63), 128 | (c & 63)); }
    }
    return new Uint8Array(arr);
  }
  function writeInt(buf, offset, val, n) {
    for (var i = 0; i < n; i++) { buf[offset + i] = (val >> (8 * i)) & 0xff; }
  }
  function crc32(data) {
    var table = crc32._t;
    if (!table) {
      table = new Uint32Array(256);
      for (var i = 0; i < 256; i++) {
        var c = i;
        for (var k = 0; k < 8; k++) { c = (c & 1) ? (0xEDB88320 ^ (c >>> 1)) : (c >>> 1); }
        table[i] = c;
      }
      crc32._t = table;
    }
    var crc = 0xFFFFFFFF;
    for (var j = 0; j < data.length; j++) { crc = table[(crc ^ data[j]) & 0xff] ^ (crc >>> 8); }
    return (crc ^ 0xFFFFFFFF) >>> 0;
  }

  var localHeaders = [];
  var centralDirs = [];
  var offset = 0;

  for (var name in files) {
    var content = files[name];
    var nameBytes = toBytes(name);
    var dataBytes = toBytes(content);
    var crc = crc32(dataBytes);

    var local = new Uint8Array(30 + nameBytes.length);
    writeInt(local, 0, 0x04034b50, 4);
    writeInt(local, 4, 20, 2);
    writeInt(local, 6, 0, 2);
    writeInt(local, 8, 0, 2);
    writeInt(local, 10, 0, 2);
    writeInt(local, 12, 0, 2);
    writeInt(local, 14, crc, 4);
    writeInt(local, 18, dataBytes.length, 4);
    writeInt(local, 22, dataBytes.length, 4);
    writeInt(local, 26, nameBytes.length, 2);
    writeInt(local, 28, 0, 2);
    local.set(nameBytes, 30);

    var central = new Uint8Array(46 + nameBytes.length);
    writeInt(central, 0, 0x02014b50, 4);
    writeInt(central, 4, 20, 2);
    writeInt(central, 6, 20, 2);
    writeInt(central, 8, 0, 2);
    writeInt(central, 10, 0, 2);
    writeInt(central, 12, 0, 2);
    writeInt(central, 14, 0, 2);
    writeInt(central, 16, crc, 4);
    writeInt(central, 20, dataBytes.length, 4);
    writeInt(central, 24, dataBytes.length, 4);
    writeInt(central, 28, nameBytes.length, 2);
    writeInt(central, 30, 0, 2);
    writeInt(central, 32, 0, 2);
    writeInt(central, 34, 0, 2);
    writeInt(central, 36, 0, 2);
    writeInt(central, 38, 0, 4);
    writeInt(central, 42, offset, 4);
    central.set(nameBytes, 46);

    localHeaders.push({ local: local, data: dataBytes, offset: offset });
    centralDirs.push(central);
    offset += local.length + dataBytes.length;
  }

  var centralSize = centralDirs.reduce(function(s, c) { return s + c.length; }, 0);
  var eocd = new Uint8Array(22);
  writeInt(eocd, 0, 0x06054b50, 4);
  writeInt(eocd, 4, 0, 2);
  writeInt(eocd, 6, 0, 2);
  writeInt(eocd, 8, localHeaders.length, 2);
  writeInt(eocd, 10, localHeaders.length, 2);
  writeInt(eocd, 12, centralSize, 4);
  writeInt(eocd, 16, offset, 4);
  writeInt(eocd, 20, 0, 2);

  var total = offset + centralSize + 22;
  var zip = new Uint8Array(total);
  var pos = 0;
  for (var h = 0; h < localHeaders.length; h++) {
    zip.set(localHeaders[h].local, pos); pos += localHeaders[h].local.length;
    zip.set(localHeaders[h].data, pos); pos += localHeaders[h].data.length;
  }
  for (var cd = 0; cd < centralDirs.length; cd++) {
    zip.set(centralDirs[cd], pos); pos += centralDirs[cd].length;
  }
  zip.set(eocd, pos);

  var blob = new Blob([zip], { type: 'application/zip' });
  var url = URL.createObjectURL(blob);
  var a = document.createElement('a');
  a.href = url;
  a.download = 'cloudshare_analytics_' + new Date().toISOString().slice(0, 10) + '.zip';
  document.body.appendChild(a);
  a.click();
  document.body.removeChild(a);
  setTimeout(function() { URL.revokeObjectURL(url); }, 5000);
}

// Auto-discover fields from dashboard 63d928c2edbb5ec6d88f930b
// ====================================================================
function exportEnvironments() {
  console.log('%c[5/6] Discovering Environments fields from dashboard...', 'color:#F05C37;font-weight:bold');
  return http.get('/api/dashboards/63d928c2edbb5ec6d88f930b/widgets')
    .then(function(resp) {
      var widgets = resp.data || [];
      var seen = {};
      var fields = [];
      for (var w = 0; w < widgets.length; w++) {
        var widget = widgets[w];
        var metadata = (widget.metadata && widget.metadata.panels) ? widget.metadata.panels : [];
        for (var p = 0; p < metadata.length; p++) {
          var items = metadata[p].items || [];
          for (var it = 0; it < items.length; it++) {
            var jaql = items[it].jaql;
            if (!jaql || !jaql.dim || !jaql.table) continue;
            if (jaql.table !== 'CloudShareEnvironmentsCube' && jaql.context) continue;
            var key = jaql.dim;
            if (!seen[key]) {
              seen[key] = true;
              fields.push({
                header: jaql.title || (jaql.table + '.' + jaql.column) || jaql.dim,
                jaql: jaql
              });
            }
          }
        }
      }
      if (fields.length === 0) {
        console.warn('    No fields discovered for Environments cube, trying fallback fields...');
        // Fallback: common environment fields
        fields = [
          { header: 'Env Name', jaql: { table: 'Environments', column: 'EnvName', dim: '[Environments.EnvName]', datatype: 'text' } },
          { header: 'Project Name', jaql: { table: 'Environments', column: 'ProjectName', dim: '[Environments.ProjectName]', datatype: 'text' } },
          { header: 'Blueprint', jaql: { table: 'Environments', column: 'Blueprint', dim: '[Environments.Blueprint]', datatype: 'text' } },
          { header: 'Status', jaql: { table: 'Environments', column: 'Status', dim: '[Environments.Status]', datatype: 'text' } },
          { header: 'Region', jaql: { table: 'Environments', column: 'Region', dim: '[Environments.Region]', datatype: 'text' } },
          { header: 'Created', jaql: { table: 'Environments', column: 'CreateTime', dim: '[Environments.CreateTime (Calendar)]', datatype: 'datetime' } }
        ];
      }
      console.log('%c    Discovered ' + fields.length + ' Environments fields', 'color:#10b981');
      return jaqlQuery('CloudShareEnvironmentsCube', fields.map(function(f) { return { jaql: f.jaql }; }));
    })
    .then(function(resp) {
      if (!resp) return;
      var values = (resp.data && resp.data.values) ? resp.data.values : [];
      console.log('%c    Environments: ' + values.length + ' rows', 'color:#10b981');
      // Rebuild field headers from resp metadata if available
      var headers = (resp.config && resp.config.data && resp.config.data.metadata)
        ? resp.config.data.metadata.map(function(m) { return m.jaql.title || m.jaql.column || m.jaql.dim; })
        : ['(fields)'];
      var rows = values.map(function(row) {
        return row.map(function(cell) { return cellVal(cell); });
      });
      results['environments.csv'] = buildCSV(headers, rows);
    })
    .catch(function(err) {
      console.error('    Environments export failed:', err);
      errors.push('CloudShareEnvironmentsCube (environments.csv)');
    });
}

          { header: 'Project Name', jaql: { table: 'ProjectMembers', column: 'ProjectName', dim: '[ProjectMembers.ProjectName]', datatype: 'text' } },
          { header: 'User Email', jaql: { table: 'ProjectMembers', column: 'UserEmail', dim: '[ProjectMembers.UserEmail]', datatype: 'text' } },
          { header: 'Role', jaql: { table: 'ProjectMembers', column: 'Role', dim: '[ProjectMembers.Role]', datatype: 'text' } },
          { header: 'User Hours', jaql: { table: 'ProjectMembers', column: 'UserHourAmount', dim: '[ProjectMembers.UserHourAmount]', datatype: 'numeric', agg: 'sum' } }
        ];
      }
      console.log('%c    Discovered ' + fields.length + ' Project Members fields', 'color:#10b981');
      return jaqlQuery('ProjectMemberUsageCube', fields.map(function(f) { return { jaql: f.jaql }; }));
    })
    .then(function(resp) {
      if (!resp) return;
      var values = (resp.data && resp.data.values) ? resp.data.values : [];
      console.log('%c    Project Members: ' + values.length + ' rows', 'color:#10b981');
      var headers = (resp.config && resp.config.data && resp.config.data.metadata)
        ? resp.config.data.metadata.map(function(m) { return m.jaql.title || m.jaql.column || m.jaql.dim; })
        : ['(fields)'];
      var rows = values.map(function(row) {
        return row.map(function(cell) { return cellVal(cell); });
      });
      results['project_members.csv'] = buildCSV(headers, rows);
    })
    .catch(function(err) {
      console.error('    Project Members export failed:', err);
      errors.push('ProjectMemberUsageCube (project_members.csv)');
    });
}

Troubleshooting
?

"Cannot read property '$injector'" error

You're not on the right page. Navigate to bi.cloudshare.com first (open Analytics from CloudShare Accelerate), then paste the script again.

?

A cube returns 0 rows

Your account may not have data in that cube, or the field names may differ. The export still completes — check which CSVs have content inside the ZIP.

?

Environments or Project Members shows wrong columns

The script auto-discovers fields from dashboard widget definitions. If the dashboard layout changed, the column headers may shift. The data itself is still correct.

?

Script times out on large datasets

Sisense caps responses at 50,000 rows per query. If you need more, run exports on a filtered date range or contact CloudShare support for a direct database export.
